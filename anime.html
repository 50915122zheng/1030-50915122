範例程式碼


            
        <link rel="stylesheet" href="anime.css">
        <https://cdnjs.cloudflare.com/ajax/libs/animejs/3.2.1/anime.min.js></script>
        
        // 定義動畫效果

    function animateBox() {
        anime({
            targets: '.box',
            translateX: 500,// 向右移500像素位置
            scale: 2,       //元素的大小放大到原來的 2 倍
            rotate: '1turn',//旋轉1圈
            duration: 2000, //動畫持續時間設置為 2000 毫秒（2 秒）
            easing: 'easeInOutQuad',//設置動畫的緩動效果
            complete: function(anim) {
                console.log("動畫完成");
                // 等待 2 秒後移動回去
                setTimeout(function() {
                    anime({
                        targets: '.box',
                        translateX: 0, // 回到原位置
                        scale: 1, // 回到原大小
                        rotate: '0turn', // 回到原始角度
                        duration: 2000,
                        easing: 'easeInOutQuad'
                    });
                }, 1000); // 等待 2000 毫秒（2 秒）
            }
        });
    }
